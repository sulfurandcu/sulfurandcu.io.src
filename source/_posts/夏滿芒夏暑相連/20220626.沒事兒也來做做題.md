### 第一題

#### 題目描述

張三和李四在玩一種遊戲。

首先，張三從 A 與 B 之間（包括A和B）選擇一個整數，告訴李四。

其次，李四從 C 與 D 之間（包括C和D）選擇一個整數。

兩個整數之和，如果是素數，張三贏，負責李四贏。

當兩個人都是最佳發揮時，誰會贏？

**輸入格式：**
一行四個正整數 A B C D 以空格隔開。

**輸入格式：**
如果張三贏了，輸出 X，否則輸出 Y。


#### 代碼實現

```
#include <iostream>
#include <sys/time.h>

using namespace std;

bool IsPrimeNumber(int n)
{
	if (n < 2)
	{
		return false;
	}

	int SquareRoot = 1;

	while (SquareRoot * SquareRoot < n)
	{
		SquareRoot++;
	}

	for (int i = 2; i <= SquareRoot; i++)
	{
		if (n % i == 0)
		{
			return false;
		}
	}

	return true;
}

int main(int argc, char **argv)
{
	int a = 0;
	int b = 0;
	int c = 0;
	int d = 0;
	int i = 0;
	int j = 0;
	int sum = 0;
	int x_win = 0;
	int y_win = 0;
	bool PrimeExist = false;

	int usec_bgn;
	int usec_end;
	int usec_sub;
	struct timeval tv;

	/* 等待輸入 */
	cin>>a>>b>>c>>d;
	// a = 1; b = 100; c = 2; d = 3;

	/* 參數判斷 */
	if (a <= 0 || a > 100 || b <= 0 || b > 100 || c <= 0 || c > 100 || d <= 0 || d > 100)
	{
		printf("error input! a, b, c, d should be greater than zero and less than or equal to 100! \r\n");
		return 0;
	}

	/* 參數判斷 */
	if (a > b || c > d)
	{
		printf("error input! b is bigger than a!\r\n");
		return 0;
	}

	/* 參數判斷 */
	if (c > d)
	{
		printf("error input! d is bigger than c!\r\n");
		return 0;
	}

	// printf("%d,%d,%d,%d\r\n", a, b, c, d);

	/* 起始時間 */
	gettimeofday(&tv, NULL);
	usec_bgn = (int)tv.tv_usec;
	printf("\r\n");
	printf("bgn time (second, micro-second): %ds, %dus\r\n", (int)tv.tv_sec, (int)tv.tv_usec);
	printf("\r\n");

	/* 勝負推演 */
	for (i = a; i <= b; ++i)
	{
		PrimeExist = false;

		for (j = c; j <= d; ++j)
		{
			sum = i + j;

			if (IsPrimeNumber(sum))
			{
				PrimeExist = true;
				break;
			}

		}

		if (PrimeExist)
		{
			y_win++;

			#if 1 /* 如果不想顯示中間過程，可以屏蔽下行（將 #if 1 改爲 #if 0） */
			printf("X = %d\tY = %d\tSUM  = %d \tY WIN !\r\n", i, j, sum);
			#endif
		}
		else
		{
			x_win++;

			#if 1 /* 如果不想顯示中間過程，可以屏蔽下行（將 #if 1 改爲 #if 0） */
			printf("X = %d\tY =  \tSUM != Prime \tX WIN !\r\n", i);
			#endif
		}
	}

	printf("\r\n");

	/* 比較小X和小Y總的勝負次數 */
	if (x_win > y_win)
	{
		printf("X WIN TIMES = %d, Y WIN TIMES = %d, SO X WINS FINALLY !\r\n", x_win, y_win);
	}
	else
	if (x_win < y_win)
	{
		printf("X WIN TIMES = %d, Y WIN TIMES = %d, SO Y WINS FINALLY !\r\n", x_win, y_win);
	}
	else
	{
		printf("X WIN TIMES = %d, Y WIN TIMES = %d, DRAWN GAME !\r\n", x_win, y_win);
	}

	/* 結束時間 */
	gettimeofday(&tv, NULL);
	usec_end = (int)tv.tv_usec;
	printf("\r\n");
	printf("end time (second, micro-second): %ds, %dus\r\n", (int)tv.tv_sec, (int)tv.tv_usec);
	printf("\r\n");

	/* 計算耗時 */
	if (usec_end >= usec_bgn)
	{
		usec_sub = usec_end - usec_bgn;
	}
	else
	{
		usec_sub = 1000000 - (usec_bgn - usec_end);
	}
	printf("time cost :%d us\r\n", usec_sub);
	printf("\r\n");

	return 0;
}
```

### 第二題

#### 題目描述

張三成爲了一家餐廳的採購員。作爲一個採購員要有良好的職業素養。必須保證每次採購的物資不超過自己車輛的容量，然後在此條件下使得餐廳賺取的收益最大。

商場中共有 n 種物資，編號爲 1 到 n 。每種物資都有兩個參數 v 和 w，分別表示這種物資的體積和採購這種物資可以給餐廳帶來的收益。注意每種物資每天至多隻能採購一件。

張三有一輛容量爲 V 的採購車，每天張三隻去採購一次，採購的物資體積之和不能超過車輛的容量。

對於每天採購的物資，餐廳老闆還有特殊的要求，第 i 天老闆要求張三不能採購編號大於等於 Li 並且小於等於 Ri 的物資（即不能採購編號在 [Li,Ri] 之間的物資）。特別的，當 Li>Ri 時表示這天沒有不可採購的物資。

對於接下來的 m 天，你需要告訴張三他每天分別最多可以給餐廳賺取多少的收益。

**輸入格式：**
輸入第一行兩個個整數 n, V 表示共有 n 種物資，張三的車子容量是 V。
接下來 n 行，每行兩個整數。
第 i+1 行輸入vi, wi 表示第 i 種物資的體積和收益。
接下來一行輸入一個整數 m ，表示需要採購的天數。
接下來 m 行，每行輸入兩個整數 Li, Ri 表示第 i 天不能採購編號位於 Li 到 Ri 之間的物資。

**輸出格式：**
輸出共 m 行，每行一個整數表示張三第 i 天可以給餐廳帶來的收益。

**輸入樣例：**
```
6   290     物資種類，容量
85  97      1號體積，收益
80  81      2號體積，收益
83  1       3號體積，收益
79  81      4號體積，收益
89  1       5號體積，收益
92  97      6號體積，收益
5           天數
3   5       Li-Ri
1   1
1   3
5   2
3   1
```

**輸出樣例：**
```
275
259
179
275
275
```

**補充說明：**
前 20% 的數據滿足: 1<=n<=10,1<=m<=1000.
前 50% 的數據滿足: 1<=n, m, V<=300.
對於 100% 的數據滿足: 1<=n, m, V<=3000; 1<=vi<=3000; 1<=Li, Ri<=n; 1<=wi<=10^6.

**時間限制：**1s

**空間限制：**512M


#### 代碼實現

```
#include <iostream>
#include <sys/time.h>

using namespace std;

int main(int argc, char **argv)
{
	int daytotal;                       // 天數
	int category;                       // 種類
	int capacity;                       // 容量
	int capacity_left;                  // 容量（剩餘）
	int edge[3001][2];                  // 邊界條件
	int menu[3001][2];                  // 貨單
	int menu_sort[3001][3];             // 貨單（根據每件貨物的收益率排序）
	int idx = 0;
	int sum = 0;                        // 收益
	double rate_max = 0;                // 收益率
	double rate_tmp = 0;                // 收益率

	int usec_bgn;
	int usec_end;
	int usec_sub;
	struct timeval tv;

	#if 0
	cin>>category>>capacity;

	for (int i = 0; i < category; ++i)
	{
		cin>>menu[i][0]>>menu[i][1];
	}

	cin>>daytotal;

	for (int i = 0; i < daytotal; ++i)
	{
		cin>>edge[i][0]>>edge[i][1];
	}
	#else
	category = 10;
	capacity = 300;
	menu[0][0] = 85; menu[0][1] = 97;
	menu[1][0] = 80; menu[1][1] = 81;
	menu[2][0] = 83; menu[2][1] = 1;
	menu[3][0] = 79; menu[3][1] = 81;
	menu[4][0] = 89; menu[4][1] = 1;
	menu[5][0] = 92; menu[5][1] = 97;
	daytotal = 10;
	edge[0][0] = 3; edge[0][1] = 5;
	edge[1][0] = 1; edge[1][1] = 1;
	edge[2][0] = 1; edge[2][1] = 3;
	edge[3][0] = 5; edge[3][1] = 2;
	edge[4][0] = 3; edge[4][1] = 1;
	#endif

	/* 起始時間 */
	gettimeofday(&tv, NULL);
	usec_bgn = (int)tv.tv_usec;
	printf("\r\n");
	printf("bgn time (second, micro-second): %ds, %dus\r\n", (int)tv.tv_sec, (int)tv.tv_usec);
	printf("\r\n");

	for (int j = 0; j < category; ++j)
	{
		idx = 0;
		rate_max = 0;

		for (int i = 0; i < category; ++i)
		{
			if (menu[i][0] == 0)
			{
				continue;
			}

			rate_tmp = (double)menu[i][1] / (double)menu[i][0];

			if (rate_tmp > rate_max)
			{
				rate_max = rate_tmp;
				idx = i;
			}
		}

		menu_sort[j][0] = menu[idx][0];
		menu_sort[j][1] = menu[idx][1];
		menu_sort[j][2] = idx;
		menu[idx][0] = 0;
		menu[idx][1] = 0;
	}

	for (int day_idx = 0; day_idx < daytotal; ++day_idx) /* 第n天 */
	{
		sum = 0;
		capacity_left = capacity;

		for (int i = 0; i < category; ++i)
		{
			if ((menu_sort[i][2]+1) >= edge[day_idx][0] && (menu_sort[i][2]+1) <= edge[day_idx][1])
			{
				continue;
			}

			if (capacity_left >= menu_sort[i][0])
			{
				capacity_left -= menu_sort[i][0];
				sum += menu_sort[i][1];
			}
		}

		printf("%d\r\n", sum);
	}

	/* 結束時間 */
	gettimeofday(&tv, NULL);
	usec_end = (int)tv.tv_usec;
	printf("\r\n");
	printf("end time (second, micro-second): %ds, %dus\r\n", (int)tv.tv_sec, (int)tv.tv_usec);
	printf("\r\n");

	/* 計算耗時 */
	if (usec_end >= usec_bgn)
	{
		usec_sub = usec_end - usec_bgn;
	}
	else
	{
		usec_sub = 1000000 - (usec_bgn - usec_end);
	}
	printf("time cost :%d us\r\n", usec_sub);
	printf("\r\n");

	return 0;
}
```
